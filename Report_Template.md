def ring_potential_point(x: float, y: float, z: float, a: float = 1.0, q: float = 1.0, n_phi: int = 720) -> float:    """    计算均匀带电圆环在空间某点产生的电势，包含数值稳定性处理    """    # 生成离散角度    phi = np.linspace(0, 2 * np.pi, n_phi,     endpoint=False)    d_phi = 2 * np.pi / n_phi        # 圆环上电荷元的位置    x_ring = a * np.cos(phi)    y_ring = a * np.sin(phi)    z_ring = 0.0        # 计算场点到各电荷元的距离，添加最小距离以增强稳定性    min_distance = 1e-10    distances = np.sqrt((x - x_ring)**2 + (y - y_ring)    **2 + (z - z_ring)**2)    distances = np.maximum(distances, min_distance)        # 计算积分    integrand = 1.0 / distances    potential = (q / (2 * np.pi)) * np.sum(integrand) *     d_phi        return float(potential)

- 物理基础 ：圆环位于xy平面，圆心在原点，半径为a，电荷均匀分布。绕z轴旋转任意角度后，圆环的位置和电荷分布保持不变。
- 等势线特征 ：在yz平面（x=0）上，等势线关于z轴对称。即对于任意点 [ o bj ec tO bj ec t ] ( y , z ) ，其关于z轴的对称点 [ o bj ec tO bj ec t ] ( − y , z ) 具有相同的电势。
- 表现形式 ：等势线在yz平面上呈现为以z轴为对称轴的闭合曲线。 2. 镜像对称性（关于xy平面）
- 物理基础 ：圆环位于xy平面，电荷分布关于xy平面对称。对于任意点 [ o bj ec tO bj ec t ] ( x , y , z ) ，其关于xy平面的镜像点 [ o bj ec tO bj ec t ] ( x , y , − z ) 到圆环各电荷元的距离相等。
